pipeline {
    agent any

    stages {
        stage('Checkout Code') {
            steps {
                echo 'Checking out from GitHub...'
                git branch: 'main', url: 'https://github.com/WaseemHajwane/dot-net.git'
            }
        }
        stage('Restore Dependencies') {
            steps {
                echo 'Restoring dependencies...'
                bat 'dotnet restore TestProject/TestProject.sln'
            }
        }
        stage('Build') {
            steps {
                echo 'Building the solution...'
                bat 'dotnet build TestProject/TestProject.sln --configuration Release'
            }
        }
        stage('Publish') {
            steps {
                echo 'Publishing the application to the existing path...'
                bat 'dotnet publish TestProject/TestProject.sln 
                --configuration Release --output D:\\DemoNetProject\\TestProject\\TestProject\\bin\\Release\\net8.0\\publish'
            }
        }
        stage('Deploy to IIS') {
            steps {
                echo 'Deploying the application to IIS...'
                bat '''
                REM Clear existing site files
                rmdir /S /Q "C:\\inetpub\\wwwroot"
                
                REM Copy published files to IIS site folder
                xcopy /E /Y /I "D:\\DemoNetProject\\TestProject\\TestProject\\bin\\Release\\net8.0\\publish" "C:\\inetpub\\wwwroot\\YourAppFolder"
                '''
            }
        }
    }

    post {
        success {
            echo 'Pipeline executed successfully, application is deployed to IIS.'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
